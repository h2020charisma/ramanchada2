377    def safe_divide(self, spe_to_correct, spe_reference_resampled):
378        numerator = spe_to_correct.y
379        # numerator_noise = spe_to_correct.y_noise
380
381        scaling_denominator = spe_reference_resampled.y / self.ref.Y(spe_reference_resampled.x)
382        print(np.median(scaling_denominator), np.mean(scaling_denominator), np.std(scaling_denominator))
383
384        # denominator_noise = spe_reference_resampled.y_noise
385        denominator = spe_reference_resampled.y
386        # Create a mask for dividing only where value is above noise !
387        # mask = (abs(scaling_denominator) > 0) & (kind_of_snr > 0.9)
388        # mask =  (abs(denominator) > abs(denominator_noise)) &
389        mask = (abs(scaling_denominator) > 0) & (numerator > 0) & (denominator > 0)
390        # & (abs(numerator) > numerator_noise) & (abs(scaling_denominator) > 0)

392        result = np.zeros_like(numerator)

405        Y = self.ref.Y(spe_reference_resampled.x)
406        mask = self.safe_mask(spe_to_correct, spe_reference_resampled)
407        if mask is None:
408            scaling_factor = Y / spe_reference_resampled.y
409        else:
410            scaling_factor = np.zeros_like(spe_reference_resampled.y)
411            scaling_factor[mask] = Y[mask] / spe_reference_resampled.y[mask]

413        result = numerator * scaling_factor
